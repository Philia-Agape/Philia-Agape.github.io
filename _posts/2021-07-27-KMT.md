pseudo-code for KMT algorithm: find substring from a string by dynamic programming
  

  CODE FOLLOWS:  
  `public class KMP {  
    private int[][] dp;  
    private String pat;  

    public KMP(String pat) {  
        this.pat = pat;  
        int M = pat.length();  
        //dp[pos][char] pos:index of string  
        //256 character  
        dp = new int[M][256];   
    
        // base case  
        dp[0][pat.charAt(0)] = 1;  
        int X = 0;  
        for (int j = 1; j < M; j++) {  
            for (int c = 0; c < 256; c++) {  
                dp[j][c] = dp[X][c];  
                //  
            dp[j][pat.charAt(j)] = j + 1;    
            // 更新影子状态  
            X = dp[X][pat.charAt(j)];    
        }  
    }  
        
    public int search(String txt) {    
        int M = pat.length();    
        int N = txt.length();    
        // j labels index of pat, starting @ 0    
        int j = 0;    
        for (int i = 0; i < N; i++) {    
            j = dp[j][txt.charAt(i)];    
            // return the start postion of the substring!    
            if (j == M) return i - M + 1;    
        }  
        return -1;  
    }  
}  `
